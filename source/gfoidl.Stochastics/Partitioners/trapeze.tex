\lambda := \frac{B-A}{N} \qquad \Rightarrow \quad X = A + \lambda \cdot n \\ \\
A_p = \frac{A+X}{2} \cdot n \qquad \Leftrightarrow \quad 2A_p = (A+X) \cdot n = (2A + \lambda \cdot n) \cdot n = 2An + \lambda \cdot n^2 \\ \\
\lambda \cdot n^2 + 2A n - 2A_p = 0 \\ \\ 
n^2 + \frac{2A}{\lambda} \cdot n - \frac{2A_p}{\lambda} = 0 \qquad \bigg \rvert \quad p:= \frac{2A}{\lambda} \quad ; \quad q:=-\frac{2A_p}{\lambda} \\ \\
n = -\frac{p}{2} \pm \sqrt{\left( \frac{p}{2} \right )^2 - q} \qquad \Rightarrow \boxed{ n (A_p) = -\frac{A}{\lambda} \pm \sqrt{\left( \frac{A}{\lambda} \right )^2 + \frac{2}{\lambda} \cdot A_p}}
